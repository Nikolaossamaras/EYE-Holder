#include <Wire.h>
#include <LiquidCrystal_I2C.h>

LiquidCrystal_I2C lcd(0x27, 16, 2); // try 0x3F if 0x27 doesn't work

unsigned long startMillis;
int startHour = 13, startMinute = 8, startSecond = 0;

void setup() {
  lcd.init();
  lcd.backlight();
  startMillis = millis();
}

void loop() {
  unsigned long elapsedSeconds = (millis() - startMillis) / 1000;

  int seconds = (startSecond + elapsedSeconds) % 60;
  int totalMinutes = startMinute + (startSecond + elapsedSeconds) / 60;
  int minutes = totalMinutes % 60;
  int hours = (startHour + totalMinutes / 60) % 24;

  lcd.setCursor(0, 0);
  lcd.print("Time:");

  lcd.setCursor(0, 1);
  if (hours < 10) lcd.print("0");
  lcd.print(hours);
  lcd.print(":");
  if (minutes < 10) lcd.print("0");
  lcd.print(minutes);
  lcd.print(":");
  if (seconds < 10) lcd.print("0");
  lcd.print(seconds);

  delay(1000);
}
